use miBaseDeDatos;

// Colección de Juego
db.createCollection("juego", {
    validator: {
        $jsonSchema: {
            bsonType: "object",
            required: ["idJuego", "tipoJuego", "nombre", "descripcion"],
            properties: {
                idJuego: { bsonType: "string" },
                tipoJuego: { bsonType: "string" },
                nombre: { bsonType: "string" },
                descripcion: { bsonType: "string" }
            }
        }
    }
});

// Colección de Empleado
db.createCollection("empleado", {
    validator: {
        $jsonSchema: {
            bsonType: "object",
            required: ["idEmpleado", "cedEmpleado", "nombre", "apellido", "telefono", "correo", "salario"],
            properties: {
                idEmpleado: { bsonType: "string" },
                cedEmpleado: { bsonType: "string" },
                nombre: { bsonType: "string" },
                apellido: { bsonType: "string" },
                telefono: { bsonType: "string" },
                correo: { bsonType: "string" },
                salario: { bsonType: "decimal" }
            }
        }
    }
});

// Colección de MaquinaExpendedora
db.createCollection("maquinaExpendedora", {
    validator: {
        $jsonSchema: {
            bsonType: "object",
            required: ["numSerie", "capacidad", "montoDisponible", "tipoMaquina"],
            properties: {
                numSerie: { bsonType: "string" },
                capacidad: { bsonType: "int" },
                montoDisponible: { bsonType: "decimal" },
                tipoMaquina: { bsonType: "string" },
                operadorxmaquina: {
                    bsonType: "array",
                    items: {
                        bsonType: "object",
                        required: ["fecha", "costo", "descripcion"],
                        properties: {
                            fecha: { bsonType: "date" },
                            costo: { bsonType: "int" },
                            descripcion: { bsonType: "string" },
                            operador: {
                                bsonType: "object",
                                required: ["idOperator", "nombre", "nombreEmpresa"],
                                properties: {
                                    idOperator: { bsonType: "string" },
                                    nombre: { bsonType: "string" },
                                    nombreEmpresa: { bsonType: "string" }
                                }
                            }
                        }
                    }
                }
            }
        }
    }
});

// Colección de Cliente
db.createCollection("cliente", {
    validator: {
        $jsonSchema: {
            bsonType: "object",
            required: ["idCliente", "cedCliente", "nombre", "apellido", "telefono", "correo", "edad"],
            properties: {
                idCliente: { bsonType: "string" },
                cedCliente: { bsonType: "string" },
                nombre: { bsonType: "string" },
                apellido: { bsonType: "string" },
                telefono: { bsonType: "string" },
                correo: { bsonType: "string" },
                edad: { bsonType: "int" },
                sesionJuego: {
                    bsonType: "array",
                    items: {
                        bsonType: "object",
                        required: ["idJuego", "fecha", "horaInicio", "horaFinal", "idEmpleado"],
                        properties: {
                            idJuego: { bsonType: "string" },
                            fecha: { bsonType: "date" },
                            horaInicio: { bsonType: "string" },
                            horaFinal: { bsonType: "string" },
                            idEmpleado: { bsonType: "string" }, // Foreign key to empleado
                            idTorneo: { bsonType: ["string", "null"] }, // Foreign key to torneo, allows null
                            apuesta: {
                                bsonType: "array",
                                items: {
                                    bsonType: "object",
                                    required: ["idApuesta", "montoApuesta", "fechaHora", "resultado", "montoGanancia"],
                                    properties: {
                                        idApuesta: { bsonType: "string" },
                                        montoApuesta: { bsonType: "int" },
                                        fechaHora: { bsonType: "date" },
                                        resultado: { bsonType: "string" },
                                        montoGanancia: { bsonType: "int" }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
    }
});

// Colección de MetodoPago
db.createCollection("metodoPago", {
    validator: {
        $jsonSchema: {
            bsonType: "object",
            required: ["idPago", "tipoPago", "datosPago"],
            properties: {
                idPago: { bsonType: "string" },
                tipoPago: { bsonType: "string" },
                datosPago: { bsonType: "string" }
            }
        }
    }
});

// Colección de TransaccionToken
db.createCollection("transaccionToken", {
    validator: {
        $jsonSchema: {
            bsonType: "object",
            required: ["idTransaccion", "tipoTransaccion", "cantidad", "fechaHora", "idMetodoPago", "idEmpleado"],
            properties: {
                idTransaccion: { bsonType: "string" },
                tipoTransaccion: { bsonType: "string" },
                cantidad: { bsonType: "int" },
                fechaHora: { bsonType: "date" },
                idMetodoPago: { bsonType: "string" }, // Foreign key to metodoPago
                idEmpleado: { bsonType: "string" }, // Foreign key to empleado
                token: {
                    bsonType: "array",
                    required: ["idToken", "valor", "tipoToken"],
                    properties: {
                        idToken: { bsonType: "string" },
                        valor: { bsonType: "int" },
                        tipoToken: { bsonType: "string" }
                    }
                }
            }
        }
    }
});

// Colección de Casino
db.createCollection("casino", {
    validator: {
        $jsonSchema: {
            bsonType: "object",
            required: ["nitCasino", "ubicacion", "capacidadClientes"],
            properties: {
                nitCasino: { bsonType: "string" },
                ubicacion: { bsonType: "string" },
                capacidadClientes: { bsonType: "int" },
                juego: {
                    bsonType: "array",
                    items: {
                        bsonType: "string" // Foreign key to juego
                    }
                },
                torneo: {
                    bsonType: "array",
                    items: {
                        bsonType: "object",
                        required: ["idTorneo", "premio", "fecha", "estado", "capacidad", "numParticipantes"],
                        properties: {
                            idTorneo: { bsonType: "string" },
                            premio: { bsonType: "decimal" },
                            fecha: { bsonType: "date" },
                            estado: { bsonType: "string" },
                            capacidad: { bsonType: "int" },
                            numParticipantes: { bsonType: "int" }
                        }
                    }
                },
                gerente: {
                    bsonType: "object",
                    required: ["empleadoID", "idGerente", "nombre", "apellido"],
                    properties: {
                        empleadoID: { bsonType: "string" }, // Foreign key to empleado
                        idGerente: { bsonType: "string" },
                        nombre: { bsonType: "string" },
                        apellido: { bsonType: "string" }
                    }
                }
            }
        }
    }
});

// Colección de CompraTicket
db.createCollection("compraTicket", {
    validator: {
        $jsonSchema: {
            bsonType: "object",
            required: ["idCliente", "fechaEntrada", "fechaSalida", "total", "estado", "placaVehiculo", "idMetodoPago", "idEmpleado"],
            properties: {
                idCliente: { bsonType: "string" }, // Foreign key to cliente
                fechaEntrada: { bsonType: "date" },
                fechaSalida: { bsonType: ["date", "null"] },
                total: { bsonType: "int" },
                tipoVehiculo: {bsonType: "string"},
                estado: { bsonType: "string" },
                placaVehiculo: { bsonType: "string" },
                idMetodoPago: { bsonType: "string" }, // Foreign key to metodoPago
                idEmpleado: { bsonType: "string" }, // Foreign key to empleado
                parqueadero: {
                    bsonType: "object",
                    required: ["capacidad", "carrosLibre", "motosLibre"],
                    properties: {
                        capacidad: { bsonType: "int" },
                        carrosLibre: { bsonType: "int" },
                        motosLibre: { bsonType: "int" }
                    }
                }
            }
        }
    }
});

// Colección de ReservarHabitacion
db.createCollection("reservarHabitacion", {
    validator: {
        $jsonSchema: {
            bsonType: "object",
            required: ["idCliente", "fechaEntrada", "fechaSalida", "descripcion", "tarifaxproductoBarato", "tarifaxproductoMedio", "tarifaxproductoCaro", "total", "idMetodoPago", "idEmpleado"],
            properties: {
                idCliente: { bsonType: "string" }, // Foreign key to cliente
                fechaEntrada: { bsonType: "date" },
                fechaSalida: { bsonType: "date" },
                descripcion: { bsonType: "string" },
                tarifaxproductoBarato: { bsonType: "int" },
                tarifaxproductoMedio: { bsonType: "int" },
                tarifaxproductoCaro: { bsonType: "int" },
                total: { bsonType: "int" },
                idMetodoPago: { bsonType: "string" }, // Foreign key to metodoPago
                idEmpleado: { bsonType: "string" }, // Foreign key to empleado
                habitacion: {
                    bsonType: "object",
                    required: ["idHabitacion", "capacidad", "tipo", "estado"],
                    properties: {
                        idHabitacion: { bsonType: "string" },
                        capacidad: { bsonType: "int" },
                        tipo: { bsonType: "string" },
                        estado: { bsonType: "string" }
                    }
                }
            }
        }
    }
});

// Índices para las foreign keys (opcional pero recomendable para el rendimiento)
db.transaccionToken.createIndex({ idMetodoPago: 1 });
db.transaccionToken.createIndex({ idEmpleado: 1 });
db.cliente.createIndex({ idEmpleado: 1 });
db.cliente.createIndex({ idTorneo: 1 });
db.cliente.createIndex({ idJuego: 1 });
db.compraTicket.createIndex({ idCliente: 1 });
db.compraTicket.createIndex({ idMetodoPago: 1 });
db.compraTicket.createIndex({ idEmpleado: 1 });
db.reservarHabitacion.createIndex({ idCliente: 1 });
db.reservarHabitacion.createIndex({ idMetodoPago: 1 });
db.reservarHabitacion.createIndex({ idEmpleado: 1 });

print("Colecciones y esquemas creados exitosamente.");
